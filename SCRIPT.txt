CREATE VIEW [dbo].[запись журнала]
AS
SELECT        dbo.Запись.ИД_Записи, dbo.Клиент.Номер_авто, dbo.Машина.Марка, dbo.Машина.Модель, dbo.Машина.Класс, dbo.Работник.Фамилия, dbo.Работник.Имя, dbo.Услуги.Наименование, dbo.Услуги.Цена
FROM            dbo.Клиент INNER JOIN
                         dbo.Запись ON dbo.Клиент.ИД_Клиента = dbo.Запись.ИД_Клиента INNER JOIN
                         dbo.Машина ON dbo.Запись.ИД_Машины = dbo.Машина.ИД_Машины INNER JOIN
                         dbo.Работник ON dbo.Запись.ИД_Работнка = dbo.Работник.ИД_Работника INNER JOIN
                         dbo.Услуги ON dbo.Запись.ИД_Услуги = dbo.Услуги.ИД_Услуги CROSS JOIN
                         dbo.[Журнал операции]
GO

                         dbo.Услуги ON dbo.Запись.ИД_Услуги = dbo.Услуги.ИД_Услуги CROSS JOIN
                         dbo.[Журнал операции]
